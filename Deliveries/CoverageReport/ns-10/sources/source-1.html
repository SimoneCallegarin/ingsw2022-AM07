437      * @param userChoice is the input string

443         if (expertMode && !characterActivated) {

445                 System.out.println("> You are logging out... are you sure? [y/n]");

448                 if (answer.equalsIgnoreCase("y")) {

456                 System.out.println(cliDrawer.drawCharacterCardsEffects());
457                 statusFlow = CLIConstants.RESTART_TASK;
458             }
459             else if (userChoice.equalsIgnoreCase("a")) {
460                 if (planning) {
461                     System.out.println("You are in the Assistant Card Phase. You can't play a character card now!");
462                     statusFlow = CLIConstants.RESTART_TASK;
463                 }
464                 else {

467                 }
468             }
469             else
470                 statusFlow = CLIConstants.CONTINUE_TASK;
471         }
472         else {
473             if (userChoice.equalsIgnoreCase("l")) {
474                 System.out.println("> You are logging out... are you sure? [y/n]");
475                 System.out.println("> ");
476                 String answer = readUserInput();
477                 if (answer.equalsIgnoreCase("y")) {
