597      * Updates when a player has the right to play according to the game phase.
598      * @param gamePhase used in order to execute different statements.
599      */
600     private void updateOrder(GamePhases gamePhase) {
601         int i = firstPlayerIndex;
602         if (gamePhase == GamePhases.PLANNING_PHASE) {
603             for (playerCounter = 0; playerCounter < numberOfPlayers; playerCounter++) {
604                 players.get(i).setOrder(CurrentOrder.getCurrentOrder(playerCounter));
605                 i++;
606                 if (i == numberOfPlayers)
607                     i = 0;
608             }
609             currentActivePlayer = CurrentOrder.FIRST_PLAYER;
610             playerCounter = 0;
611         }
612 
613         if (gamePhase == GamePhases.ACTION_PHASE) {
614             TreeSet<Integer> turnOrderTree = new TreeSet<>();
615             int lowestTurnOrder = 0;
616 
617             for (Player p : players)
618                 turnOrderTree.add(Integer.valueOf(Integer.valueOf(p.getDiscardPile().getTurnOrder()).toString()+Integer.valueOf(p.getCardOrder()).toString()));
619 
620             for (playerCounter = 0; playerCounter < numberOfPlayers; playerCounter++) {
621                 if (!turnOrderTree.isEmpty())
622                     lowestTurnOrder = turnOrderTree.pollFirst();
623                 for (Player p : players) {
624                     if (Integer.parseInt(Integer.valueOf(p.getDiscardPile().getTurnOrder()).toString()+Integer.valueOf(p.getCardOrder()).toString()) == lowestTurnOrder) {
625                         p.setOrder(CurrentOrder.getCurrentOrder(playerCounter));
626                         break;
627                     }
628                 }
629             }
630 
631             for (Player p : players)
632                 if (p.getOrder() == CurrentOrder.FIRST_PLAYER)
633                     firstPlayerIndex = p.getDashboard().getDashboardID();
634             currentActivePlayer = CurrentOrder.FIRST_PLAYER;
635             playerCounter = 0;
636         }
637 
638         notifyTurn();
639 
640     }

792     public void checkEndGame() {
793         for (Player p : players)
794             if (p.getDashboard().getTowerStorage().getNumberOfTowers() == 0 && p.getDashboard().getTowerStorage().getTowerColor() != TowerColors.NOCOLOR) {
795                 endGame = true;

801             endGame = true;
802 
803             int minorNumOfTowersInStorage = 8;
804             int winnerIndex = 0;
805             boolean draw = false;
806 
807             for (Player p : players) {
808                 if (p.getDashboard().getTowerStorage().getNumberOfTowers() == minorNumOfTowersInStorage && p.getDashboard().getTowerStorage().getTowerColor() != TowerColors.NOCOLOR)
809                     draw = true;

811                     minorNumOfTowersInStorage = p.getDashboard().getTowerStorage().getNumberOfTowers();
812                     winnerIndex = p.getDashboard().getDashboardID();
813                     draw = false;
814                 }
815             }
816

819

822                         drawEndGame = true;
823                     if (p.getDashboard().getDiningRoom().getNumberOfProfessors() > majorProfessors && p.getDashboard().getTowerStorage().getNumberOfTowers() == minorNumOfTowersInStorage) {
824                         majorProfessors = p.getDashboard().getDiningRoom().getNumberOfProfessors();
825                         winnerIndex = p.getDashboard().getDashboardID();
826                         drawEndGame = false;
827                     }
828                 }
829             }
830 
831             winner = players.get(winnerIndex);
832         }
